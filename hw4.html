import numpy as np
import matplotlib
import matplotlib.pyplot as plt
import scipy
from scipy import stats

t = np.linspace(0,np.pi*100,np.int(1e5), endpoint=True)
np.append(t, [0])
N = 10+np.sin(t)
np.append(N, [0])
bkgd = stats.norm.rvs(size=np.int(1e5))*np.sqrt(N)+N

def slicing(n1, n2, spacing):
    t_ = t[n1:n2:spacing]
    bkgd_ = bkgd[n1:n2:spacing]
    fig, ax = plt.subplots(1, 1)
    ax.plot(t_, bkgd_, 'o', lw=5, alpha=0.6)
def slicin(n1, n2):
    slicing(n1, n2, 1)
